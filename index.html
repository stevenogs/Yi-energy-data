class YiEnergyEngine { constructor() { // 核心定錨 this.origin_matrix = { "shen": true, "zi": true, "chen": true, "xu": true }; } _countOccurrences(arr, val) { return arr.filter(item => item === val).length; } calculateStresses(year, month, day, hour) { const solar = Solar.fromYmdHms(year, month, day, hour, 0, 0); const lunar = Lunar.fromSolar(solar); const eight = lunar.getEightChar(); // 獲取四柱 const allZhi = [eight.getYearZhi(), eight.getMonthZhi(), eight.getDayZhi(), eight.getTimeZhi()]; const allGan = [eight.getYearGan(), eight.getMonthGan(), eight.getDayGan(), eight.getTimeGan()]; const hZhi = eight.getTimeZhi(); // 流時地支 let wealthScore = 1.0; let stressScore = 1.0; let activeStatuses = []; let logs = []; // --- 【邏輯 A】水為核心 --- let waterElements = this._countOccurrences(allZhi, "子") + this._countOccurrences(allGan, "壬") + this._countOccurrences(allGan, "癸"); if (allZhi.includes("子")) { wealthScore += 2.5; activeStatuses.push("WEALTH_FLOW"); logs.push("🌊 [申子辰] 財星大潮解鎖"); } else if (waterElements > 0) { wealthScore += (waterElements * 0.8); activeStatuses.push("WEALTH_READY"); logs.push(`💰 財星浮頭：天干見 ${waterElements} 重壬/癸水`); } if (["申", "酉"].includes(hZhi)) { wealthScore += 1.0; if (!activeStatuses.includes("WEALTH_FLOW") && !activeStatuses.includes("WEALTH_READY")) { activeStatuses.push("PRODUCTION_ON"); } logs.push(`🔨 [金生水] 食傷產線送料：${hZhi} 時，生財效率倍增`); } // --- 【邏輯 B】環境熔斷 --- let xuCount = this._countOccurrences(allZhi, "戌"); if (xuCount > 0 && this.origin_matrix["chen"]) { stressScore += (xuCount * 1.5); activeStatuses.push("TURBULENCE"); logs.push(`💥 [辰戌大衝] 觸發：${xuCount} 重戌土衝擊辰土`); } if (allGan.includes("乙") && allGan.includes("庚")) { stressScore -= 1.0; wealthScore += 0.5; activeStatuses.push("STRESS_VALVE"); logs.push("🤝 [乙庚合金] 壓力分流閥啟動"); } // --- 【邏輯 C】靈魂定錨 --- if (allZhi.includes("午") && this.origin_matrix["xu"]) { wealthScore += 0.5; activeStatuses.push("ANCHORED"); logs.push("🔥 [午戌半合火局] 補能"); } if (allZhi.includes("未") && (allZhi.includes("辰") || this.origin_matrix["chen"])) { activeStatuses.push("BREAKTHROUGH"); logs.push("🔨 [辰未相破]：打破舊循環，建立新架構"); } // --- 【邏輯 D】風險控制 --- let woodCount = this._countOccurrences(allZhi, "寅") + this._countOccurrences(allZhi, "卯"); if (woodCount >= 1) { if (allZhi.includes("申") || this.origin_matrix["shen"]) { stressScore += (woodCount * 1.2); activeStatuses.push("LOW_POWER_DEFENSE"); logs.push("🚨 [強木克土] 啟動低耗能防禦"); } } return { system_status: activeStatuses.length > 0 ? activeStatuses.join(" + ") : "NORMAL", wealth_score: Math.min(Math.max(wealthScore, 1.0), 5.0).toFixed(1), stress_score: Math.min(Math.max(stressScore, 1.0), 5.0).toFixed(1), engine_logs: logs }; } }
